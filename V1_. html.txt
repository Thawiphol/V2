      async function onConfigChange(config) {
        const cfg = Object.assign({}, defaultConfig, config || {});
        const root = document.getElementById("planner-root");
        const card = document.getElementById("planner-card");
        if (!root || !card) return;

        const baseFontStack = "Arial, sans-serif";
        const fontFamily = cfg.font_family + ", " + baseFontStack;
        const baseSize = Number(cfg.font_size) || defaultConfig.font_size;

        root.style.backgroundColor = cfg.background_color;
        card.style.backgroundColor = cfg.surface_color;
        card.style.color = cfg.text_color;

        const shadowEls = ["mm-shadow","lf-shadow","tl-shadow","ph-shadow","gr-shadow"];
        shadowEls.forEach(id => {
          const el = document.getElementById(id);
          if (el) el.style.backgroundColor = cfg.secondary_action_color;
        });

        const titleBars = ["mm-title-bar","lf-title-bar","tl-title-bar","ph-title-bar","gr-title-bar"];
        titleBars.forEach(id => {
          const el = document.getElementById(id);
          if (el) {
            el.style.backgroundColor = cfg.primary_action_color;
            el.style.color = "#ffffff";
          }
        });

        const cardIds = ["mm-card","lf-card","tl-card","ph-card","gr-card"];
        cardIds.forEach(id => {
          const el = document.getElementById(id);
          if (el) {
            el.style.backgroundColor = cfg.surface_color;
            el.style.borderColor = cfg.primary_action_color;
          }
        });

        const titleEl = document.getElementById("main-title");
        const subtitleEl = document.getElementById("subtitle-text");
        const footerNote = document.getElementById("footer-note");
        if (titleEl) {
          titleEl.style.fontFamily = fontFamily;
          titleEl.style.fontSize = (baseSize * 1.6) + "px";
          titleEl.textContent = cfg.main_title;
        }
        if (subtitleEl) {
          subtitleEl.style.fontFamily = fontFamily;
          subtitleEl.style.fontSize = (baseSize * 0.85) + "px";
          subtitleEl.textContent = cfg.subtitle;
        }
        if (footerNote) {
          footerNote.style.fontFamily = fontFamily;
          footerNote.style.fontSize = (baseSize * 0.8) + "px";
          footerNote.textContent = cfg.footer_note;
        }

        const mmTitle = document.getElementById("mm-title");
        const lfTitle = document.getElementById("lf-title");
        const tlTitle = document.getElementById("tl-title");
        const phTitle = document.getElementById("ph-title");
        const grTitle = document.getElementById("gr-title");
        const mmPrompt = document.getElementById("mm-prompt");
        const grPrompt = document.getElementById("gr-prompt");

        if (mmTitle) {
          mmTitle.textContent = cfg.morning_mindset_title;
          mmTitle.style.fontFamily = fontFamily;
          mmTitle.style.fontSize = (baseSize * 0.9) + "px";
        }
        if (lfTitle) {
          lfTitle.textContent = cfg.laser_focus_title;
          lfTitle.style.fontFamily = fontFamily;
          lfTitle.style.fontSize = (baseSize * 0.9) + "px";
        }
        if (tlTitle) {
          tlTitle.textContent = cfg.task_list_title;
          tlTitle.style.fontFamily = fontFamily;
          tlTitle.style.fontSize = (baseSize * 0.9) + "px";
        }
        if (phTitle) {
          phTitle.textContent = cfg.physical_health_title;
          phTitle.style.fontFamily = fontFamily;
          phTitle.style.fontSize = (baseSize * 0.9) + "px";
        }
        if (grTitle) {
          grTitle.textContent = cfg.growth_title;
          grTitle.style.fontFamily = fontFamily;
          grTitle.style.fontSize = (baseSize * 0.9) + "px";
        }
        if (mmPrompt) {

            btn.style.backgroundColor = "transparent";
            btn.style.borderColor = cfg.secondary_action_color;

      function mapToCapabilities(config) {
        const cfg = Object.assign({}, defaultConfig, config || {});
        return {
          recolorables: [
            {
              get: () => cfg.background_color,
              set: (value) => {
                if (!window.elementSdk) return;
                window.elementSdk.setConfig({ background_color: value });
              }
            },
            {
              get: () => cfg.surface_color,
              set: (value) => {
                if (!window.elementSdk) return;
                window.elementSdk.setConfig({ surface_color: value });
              }
            },
            {
              get: () => cfg.text_color,
              set: (value) => {
                if (!window.elementSdk) return;
                window.elementSdk.setConfig({ text_color: value });
              }
            },
            {
              get: () => cfg.primary_action_color,
              set: (value) => {
                if (!window.elementSdk) return;
                window.elementSdk.setConfig({ primary_action_color: value });
              }
            },
            {
              get: () => cfg.secondary_action_color,
              set: (value) => {
                if (!window.elementSdk) return;
                window.elementSdk.setConfig({ secondary_action_color: value });
              }
            }
          ],
          borderables: [],
          fontEditable: {
            get: () => cfg.font_family,
            set: (value) => {
              if (!window.elementSdk) return;
              window.elementSdk.setConfig({ font_family: value });
            }
          },
          fontSizeable: {
            get: () => cfg.font_size,
            set: (value) => {
              if (!window.elementSdk) return;
              const size = typeof value === "number" ? value : parseFloat(value);
              if (!isNaN(size)) {
                window.elementSdk.setConfig({ font_size: size });
              }
            }
          }
        };
      }

      function mapToEditPanelValues(config) {
        const cfg = Object.assign({}, defaultConfig, config || {});
        return new Map([
          ["main_title", cfg.main_title],
          ["subtitle", cfg.subtitle],
          ["morning_mindset_title", cfg.morning_mindset_title],
          ["morning_mindset_prompt", cfg.morning_mindset_prompt],
          ["laser_focus_title", cfg.laser_focus_title],
          ["task_list_title", cfg.task_list_title],
          ["physical_health_title", cfg.physical_health_title],
          ["growth_title", cfg.growth_title],
          ["growth_prompt", cfg.growth_prompt],
          ["footer_note", cfg.footer_note]
        ]);
      }

      if (window.elementSdk) {
        window.elementSdk.init({
          defaultConfig,
          onConfigChange,
          mapToCapabilities,
          mapToEditPanelValues
        });
      } else {
        onConfigChange(defaultConfig);
      }
